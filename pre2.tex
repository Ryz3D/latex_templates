% \intropage {content}
\newcommand{\intropage}[1]{
\pagenumbering{gobble}
{
\linespread{1.5}
#1
{\let\newpage\relax\maketitle}
}
\pagebreak
\pagenumbering{arabic}
}

% \tocline {text}
\newcommand{\tocline}[1]{\addcontentsline{toc}{section}{\protect\numberline{}#1}}
% \tocsubline {text}
\newcommand{\tocsubline}[1]{\addcontentsline{toc}{subsection}{\protect\numberline{}#1}}

% \extrasection {text}
\newcommand{\extrasection}[1]{
\phantomsection
\section*{#1}
\tocline{#1}
}
% \extrasubsection {text}
\newcommand{\extrasubsection}[1]{
\phantomsection
\subsection*{#1}
\tocsubline{#1}
}

% \listfigname

\newcommand{\listfig}{

\renewcommand\listfigurename{}

}

}
