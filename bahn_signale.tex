% Symbole für Eisenbahnsignale in schematischem Signallageplan

\newcommand{\tikzSignalLabel}[2]{
	node[#1=5pt of #2,text height=28pt]{\fontsize{28pt}{28pt}\selectfont #2}
}

% Hp1 Rechts
\newcommand{\tikzSignalHpR}[1]{
	node(#1)[draw,minimum size=33]{}
	++(-0.6,0) -- ++(-2,0)
	++(0,0.5) -- ++(0,-1) ++(0,0.5)
	\tikzSignalLabel{below}{#1}
}

% Hp1 Rechts gefüllt
\newcommand{\tikzSignalHpFR}[1]{
	node(#1)[draw,minimum size=33,fill]{}
	++(-0.6,0) -- ++(-2,0)
	++(0,0.5) -- ++(0,-1) ++(0,0.5)
	\tikzSignalLabel{below}{#1}
}

% Hp1+Vr Rechts gefüllt
\newcommand{\tikzSignalHpFVR}[2]{
	node(#1)[draw,minimum size=33,fill]{}
	++(-0.6,0) -- ++(-2,0)
	++(0,0.5) -- ++(0,-1) ++(0,0.5)
	++(1.5,-0.35) node[draw,minimum size=10,fill]{}
	++(-0.5,0.7) node[draw,minimum size=10,fill]{}
	++(0.25,-0.35) node(#2)[minimum size=33]{}
	\tikzSignalLabel{xshift=30,below}{#1}
	\tikzSignalLabel{below}{#2}
}
