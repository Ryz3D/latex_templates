% Symbole für Eisenbahnsignale in schematischem Signallageplan

\newcommand{\tikzSignalLabel}[2]{
	(#2) node[#1,text height=5.5pt]{\tiny #2}
}

% Hp1 Rechts
\newcommand{\tikzSignalHpR}[1]{
	node(#1)[draw,minimum size=5]{}
	++(-0.6,0) -- ++(-2,0)
	++(0,0.5) -- ++(0,-1) ++(0,0.5)
	\tikzSignalLabel{below}{#1}
}

% Hp1 Rechts gefüllt
\newcommand{\tikzSignalHpFR}[1]{
	node(#1)[draw,minimum size=5,fill]{}
	++(-0.6,0) -- ++(-2,0)
	++(0,0.5) -- ++(0,-1) ++(0,0.5)
	\tikzSignalLabel{below}{#1}
}

% Hp1+Vr Rechts gefüllt
\newcommand{\tikzSignalHpFVR}[2]{
	node(#1)[draw,minimum size=5,fill]{}
	++(-0.6,0) -- ++(-2,0)
	++(0,0.5) -- ++(0,-1) ++(0,0.5)
	++(1.5,-0.35) node[draw,minimum size=1.5,fill]{}
	++(-0.5,0.7) node[draw,minimum size=1.5,fill]{}
	++(0.25,-0.35) node(#2)[minimum size=5]{}
	\tikzSignalLabel{below,xshift=5}{#1}
	\tikzSignalLabel{below,xshift=-5}{#2}
}
